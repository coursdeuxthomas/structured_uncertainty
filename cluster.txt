Se connecter au cluster depuis vscode :
1/ Ctrl + Shift + P ; Remote-SSH: Add New SSH Host ; ssh tbouru@ohpc.s.upf.edu
2/ Remote-SSH: Connect to Host ; 
3/ Voire si tu es connecter : 
>>> hostname
4/ Activer l'environnement : 
>>> conda activate dncnn
5/alias:
voire en direct le dernier jobs
alias lastlog='tail -f $(ls -t ~/projet_dncnn/dncnn_*.out | head -1)'
voire les jobs en cours
alias jobs='squeue -u tbouru'
voire le nom du dernier jobs 
alias lastout='ls -t ~/projet_dncnn/dncnn_*.out | head -1'
run un code train.bash
alias run='sbatch ~/train.bash'
kill tous les jobs
alias killalljobs='scancel -u tbouru'
